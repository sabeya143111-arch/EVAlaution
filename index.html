} function json_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /* ---------- data sheet ---------- */ function ensureDataSheet_(ss) { let sheet = ss.getSheetByName(SHEET_DATA); ensureDataSheet_(ss); buildSummary_(ss); buildDashboard_(ss); SpreadsheetApp.getUi && SpreadsheetApp.flush(); } /* ---------- hidden Summary sheet (feeds charts) ---------- */ function buildSummary_(ss) { let s = ss.getSheetByName(SHEET_SUM); if (!s) s = ss.insertSheet(SHEET_SUM); s.clear(); s.getCharts().forEach(c => s.removeChart(c)); // A:C → الفرع | متوسط الدرجة | عدد التقييمات s.getRange('A1').setFormula( '=IFERROR(QUERY(' + SHEET_DATA + '!A2:AF,' + '"select D, avg(F), count(F) where D is not null group by D order by avg(F) desc ' + 'label D \'الفرع\', avg(F) \'متوسط الدرجة\', count(F) \'عدد التقييمات\' format avg(F) \'0.0\'",0),"—")' ); // E:F → الكفاءة | المتوسط % const catLabels = ['التفويض والثقة','التواصل والمتابعة','اتخاذ القرار','إدارة الأداء','بناء الفريق']; const catCols = ['H','I','J','K','L']; s.getRange('E1').setValue('الكفاءة'); for (let i = 0; i < kpis.length; i++) { const col = 1 + i; d.getRange(3, col).setValue(kpis[i][0]).setFontWeight('bold').setFontColor('#6b6b80').setFontSize(11).setHorizontalAlignment('center'); d.getRange(4, col).setFormula(kpis[i][1]).setFontFamily('Cairo').setFontSize(26).setFontWeight('bold').setFontColor('#171430').setHorizontalAlignment('center');
